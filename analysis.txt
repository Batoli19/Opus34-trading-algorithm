### 🎯 Priority 3: Add Database Maintenance

In `trading_memory.py` add:
```python
def archive_old_trades(self, days_old=365):
    """Move trades older than X days to archive table"""
    cursor = self.conn.cursor()
    cutoff = datetime.now() - timedelta(days=days_old)
    
    # Copy to archive
    cursor.execute("""
        INSERT INTO trades_archive 
        SELECT * FROM trades WHERE entry_time < ?
    """, (cutoff,))
    
    # Delete from main
    cursor.execute("DELETE FROM trades WHERE entry_time < ?", (cutoff,))
    self.conn.commit()
```

Run weekly via cron job or on startup.

def calibrate_confidence(self):

    trades = self.memory.get_all_closed_trades()
    
    # Group by confidence buckets
    buckets = {
        "60-70%": [],
        "70-80%": [],
        "80-90%": [],

    }
    
    for t in trades:
        if 0.6 <= t.confidence < 0.7:
            buckets["60-70%"].append(t.outcome == "WIN")
        # ... etc
    
    # Calculate actual win rates
    for bucket, results in buckets.items():
        actual_wr = sum(results) / len(results) if results else 0
